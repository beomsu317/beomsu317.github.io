Description

<?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:app="http://schemas.android.com/apk/res-auto"
    xmlns:tools="http://schemas.android.com/tools"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    android:orientation="vertical"
    tools:context=".MainActivity">

    <TextView
        android:id="@+id/textView"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:textSize="30sp"
        android:text="Hello World!" />

    <Spinner
        android:id="@+id/spinner"
        android:layout_width="match_parent"
        android:layout_height="wrap_content" />

</LinearLayout>

MainActivity 클래스는 어댑터를 직접 정의하지 않고 기본 어댑터를 사용하여 스피너를 구현한다. 아이템이 선택됐을 경우 해당 아이템으로 textView를 변경한다.

public class MainActivity extends AppCompatActivity {

    TextView textView;
    String[] items = {"aa","bb","cc","dd","ee","ff","gg"};

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        textView = findViewById(R.id.textView);
        Spinner spinner = findViewById(R.id.spinner);
        
        // 기본 어댑터 사용
        ArrayAdapter<String> adapter = new ArrayAdapter<String>(this, android.R.layout.simple_spinner_item,items);
        // 스피너의 아이템들을 보여줄 뷰에 사용되는 레이아웃 지정
        adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        spinner.setAdapter(adapter);

        // 스피너 객체가 아이템 선택 이벤트를 처리할 수 있는 OnItemSelectedListener 리스너 
        spinner.setOnItemSelectedListener(new AdapterView.OnItemSelectedListener() {
            // 아이템이 선택되면 onItemSelected() 메서드 자동 호출
            @Override
            public void onItemSelected(AdapterView<?> parent, View view, int position, long id) {
                textView.setText(items[position]);
            }

            @Override
            public void onNothingSelected(AdapterView<?> parent) {
                textView.setText("");
            }
        });

    }
}

Conclusion

스피너를 통해 아이템이 선택될 경우 해당 아이템의 문자열이 textView에 보여지게 된다.
